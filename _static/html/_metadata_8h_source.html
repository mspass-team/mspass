413template <typename T> T Metadata::get(const std::string key) const {
414 T result;
415 std::map<std::string, boost::any>::const_iterator iptr;
416 iptr = md.find(key);
417 if (iptr == md.end()) {
418 throw MetadataGetError(key, typeid(T).name());
419 }
420 boost::any aval = iptr->second;
421 try {
422 result = boost::any_cast<T>(aval);
423 } catch (boost::bad_any_cast &err) {
424 const std::type_info &ti = aval.type();
425 throw MetadataGetError(err.what(), key, typeid(T).name(), ti.name());
426 };
427 return result;
428}
