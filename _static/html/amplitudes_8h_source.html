55 template <typename Tdata> double scale(Tdata& d,const ScalingMethod method,
56  const double level, const mspass::algorithms::TimeWindow win)
57 {
58  if((method==ScalingMethod::ClipPerc) && (level<=0.0 || level>1.0))
59  throw mspass::utility::MsPASSError("scale function: illegal perf level specified for clip percentage scale - must be between 0 and 1\nData unaltered - may cause downstream problems",
60  mspass::utility::ErrorSeverity::Suspect);
61  try{
62  double newcalib(1.0);
63  /* the else condition here should perhaps generate an elog message but
64  did not implement to allow this template to be used for CoreTimeSeries
65  and CoreSeismogram that do not have an elog attribute.*/
66  if(d.is_defined(scale_factor_key))
67  {
68  newcalib=d.get_double(scale_factor_key);
69  }
70  /* Handle time windowing. Log window mismatches but silently handle
71  cast where the window is invalid - used as a way to override any
72  time windowing */
73  mspass::algorithms::TimeWindow ampwindow;
74  if(win.start>win.end)
75  {
76  ampwindow.start=d.t0();
77  ampwindow.end=d.endtime();
78  }
79  else if( (fabs(win.start-d.t0())/d.dt()>0.5)
80  || (fabs(win.end-d.endtime())/d.dt() > 0.5) )
81  {
82  std::stringstream ss;
83  ss << "Window time range is inconsistent with input data range"<<std::endl
84  << "Input data starttime="<<d.t0()<<" and window start time="
85  << win.start <<" Difference="<<d.t0()-win.start<<std::endl
86  << "Input data endtime="<<d.endtime()<<" and window end time="
87  << win.end <<" Difference="<<d.endtime()-win.end<<std::endl
88  << "One or the other exceeds 1/sample interval="<<d.dt()<<std::endl
89  << "Window for amplitude calculation changed to data range";
90  d.elog.log_error("scale",ss.str(),mspass::utility::ErrorSeverity::Complaint);
91  ampwindow.start=d.t0();
92  ampwindow.end=d.endtime();
93  }
94  else
95  {
96  ampwindow=win;
97  }
98  Tdata windowed_data;
99  windowed_data=mspass::algorithms::WindowData(d,ampwindow);
100  double amplitude,dscale;
101  switch(method)
102  {
103  case ScalingMethod::Peak:
104  amplitude=PeakAmplitude(windowed_data);
105  break;
106  case ScalingMethod::ClipPerc:
107  amplitude=PercAmplitude(windowed_data,level);
108  break;
109  case ScalingMethod::MAD:
110  amplitude=MADAmplitude(windowed_data);
111  break;
112  case ScalingMethod::RMS:
113  default:
114  amplitude=RMSAmplitude(windowed_data);
115  };
116  /* needed to handle case with a vector of all 0s*/
117  if(amplitude>0.0)
118  {
119  dscale = level/amplitude;
120  newcalib /= dscale;
121  d *= dscale;
122  d.put(scale_factor_key,newcalib);
123  }
124  else
125  {
126  std::stringstream ss;
127  ss << "Data array is all 0s and cannot be scaled";
128  d.elog.log_error("scale",ss.str(),mspass::utility::ErrorSeverity::Complaint);
129  /* This may not be necessary but it assures this value is always set on
130  return even if it means nothing*/
131  d.put(scale_factor_key,newcalib);
132  }
133  return amplitude;
134  }catch(...){throw;};
135 }

199 template <typename Tdata> double scale_ensemble(mspass::seismic::Ensemble<Tdata>& d,
200  const ScalingMethod& method, const double level, const bool use_mean)
201 {
202  if((method==ScalingMethod::ClipPerc) && (level<=0.0 || level>1.0))
203  throw mspass::utility::MsPASSError("scale_ensemble function: illegal perf level specified for clip percentage scale - must be between 0 and 1\nData unaltered - may cause downstream problems",
204  mspass::utility::ErrorSeverity::Suspect);
205  try{
206  double avgamp; //defined here because the value computed here is returned on success
207  typename std::vector<Tdata>::iterator dptr;
208  std::vector<double> amps;
209  amps.reserve(d.member.size());
210  size_t nlive(0);
211  for(dptr=d.member.begin();dptr!=d.member.end();++dptr)
212  {
213  double amplitude;
214  if(dptr->dead()) continue;
215  switch(method)
216  {
217  case ScalingMethod::Peak:
218  amplitude=PeakAmplitude(*dptr);
219  break;
220  case ScalingMethod::ClipPerc:
221  amplitude=PercAmplitude(*dptr,level);
222  break;
223  case ScalingMethod::MAD:
224  amplitude=MADAmplitude(*dptr);
225  break;
226  case ScalingMethod::RMS:
227  default:
228  amplitude=RMSAmplitude(*dptr);
229  };
230  ++nlive;
231  amps.push_back(log(amplitude));
232  }
233  /*Silently return a 0 if there are no live data members*/
234  if(nlive==0) return 0.0;
235  mspass::utility::VectorStatistics<double> ampstats(amps);
236  if(use_mean)
237  {
238  avgamp=ampstats.mean();
239  }
240  else
241  {
242  avgamp=ampstats.median();
243  }
244 
245  /* restore to a value instead of natural log*/
246  avgamp=exp(avgamp);
247  /* Silently do nothing if all the data are zero. Would be better to
248  log an error but use as a "Core" object doesn't contain an elog attribute*/
249  if(avgamp<=0.0)
250  {
251  return 0.0;
252  }
253  double dscale=level/avgamp;
254  /* Now scale the data and apply calib */
255  for(dptr=d.member.begin();dptr!=d.member.end();++dptr)
256  {
257  if(dptr->live())
258  {
259  double calib;
260  (*dptr) *= dscale;
261  if(dptr->is_defined(scale_factor_key))
262  {
263  calib=dptr->get_double(scale_factor_key);
264  }
265  else
266  {
267  calib=1.0;
268  }
269  calib/=dscale;
270  dptr->put(scale_factor_key,calib);
271  }
272  }
273  return avgamp;
274  }catch(...){throw;};
275 }
277 template <class T> std::vector<T> normalize(const std::vector<T>& d)
278 {
279  size_t N = d.size();
280  std::vector<T> result;
281  result.reserve(N);
282  double d_nrm(0.0);;
283  for(size_t i=0;i<N;++i)
284  {
285  d_nrm += (d[i]*d[i]);
286  result.push_back(d[i]);
287  }
288  d_nrm = sqrt(d_nrm);
289  for(size_t i=0;i<N;++i) result[i] /= d_nrm;
290  return result;
