53 template <typename Tdata> double scale(Tdata& d,const ScalingMethod method,
54  const double level, const mspass::algorithms::TimeWindow win)
55 {
56  if((method==ScalingMethod::ClipPerc) && (level<=0.0 || level>1.0))
57  throw mspass::utility::MsPASSError("scale function: illegal perf level specified for clip percentage scale - must be between 0 and 1\nData unaltered - may cause downstream problems",
58  mspass::utility::ErrorSeverity::Suspect);
59  try{
60  double newcalib(1.0);
61  /* the else condition here should perhaps generate an elog message but
62  did not implement to allow this template to be used for CoreTimeSeries
63  and CoreSeismogram that do not have an elog attribute.*/
64  if(d.is_defined(scale_factor_key))
65  {
66  newcalib=d.get_double(scale_factor_key);
67  }
68  /* Handle time windowing. Log window mismatches but silently handle
69  cast where the window is invalid - used as a way to override any
70  time windowing */
71  mspass::algorithms::TimeWindow ampwindow;
72  if(win.start>win.end)
73  {
74  ampwindow.start=d.t0();
75  ampwindow.end=d.endtime();
76  }
77  else if( (fabs(win.start-d.t0())/d.dt()>0.5)
78  || (fabs(win.end-d.endtime())/d.dt() > 0.5) )
79  {
80  std::stringstream ss;
81  ss << "Window time range is inconsistent with input data range"<<std::endl
82  << "Input data starttime="<<d.t0()<<" and window start time="
83  << win.start <<" Difference="<<d.t0()-win.start<<std::endl
84  << "Input data endtime="<<d.endtime()<<" and window end time="
85  << win.end <<" Difference="<<d.endtime()-win.end<<std::endl
86  << "One or the other exceeds 1/sample interval="<<d.dt()<<std::endl
87  << "Window for amplitude calculation changed to data range";
88  d.elog.log_error("scale",ss.str(),mspass::utility::ErrorSeverity::Complaint);
89  ampwindow.start=d.t0();
90  ampwindow.end=d.endtime();
91  }
92  else
93  {
94  ampwindow=win;
95  }
96  Tdata windowed_data;
97  windowed_data=mspass::algorithms::WindowData(d,ampwindow);
98  double amplitude,dscale;
99  switch(method)
100  {
101  case ScalingMethod::Peak:
102  amplitude=PeakAmplitude(windowed_data);
103  dscale = level/amplitude;
104  newcalib /= dscale;
105  break;
106  case ScalingMethod::ClipPerc:
107  amplitude=PerfAmplitude(windowed_data,level);
108  /* for this scaling we use level as perf and output level is frozen
109  to be scaled to order unity*/
110  dscale = 1.0/amplitude;
111  newcalib /= dscale;
112  break;
113  case ScalingMethod::MAD:
114  amplitude=MADAmplitude(windowed_data);
115  dscale = level/amplitude;
116  newcalib /= dscale;
117  break;
118  case ScalingMethod::RMS:
119  default:
120  amplitude=RMSAmplitude(windowed_data);
121  dscale = level/amplitude;
122  newcalib /= dscale;
123  };
124  d *= dscale;
125  d.put(scale_factor_key,newcalib);
126  return amplitude;
127  }catch(...){throw;};
128 }

192 template <typename Tdata> double scale_ensemble(mspass::seismic::Ensemble<Tdata>& d,
193  const ScalingMethod& method, const double level, const bool use_mean)
194 {
195  if((method==ScalingMethod::ClipPerc) && (level<=0.0 || level>1.0))
196  throw mspass::utility::MsPASSError("scale_ensemble function: illegal perf level specified for clip percentage scale - must be between 0 and 1\nData unaltered - may cause downstream problems",
197  mspass::utility::ErrorSeverity::Suspect);
198  try{
199  double avgamp; //defined here because the value computed here is returned on success
200  typename std::vector<Tdata>::iterator dptr;
201  std::vector<double> amps;
202  amps.reserve(d.member.size());
203  size_t nlive(0);
204  for(dptr=d.member.begin();dptr!=d.member.end();++dptr)
205  {
206  double amplitude;
207  if(dptr->dead()) continue;
208  switch(method)
209  {
210  case ScalingMethod::Peak:
211  amplitude=PeakAmplitude(*dptr);
212  break;
213  case ScalingMethod::ClipPerc:
214  amplitude=PerfAmplitude(*dptr,level);
215  break;
216  case ScalingMethod::MAD:
217  amplitude=MADAmplitude(*dptr);
218  break;
219  case ScalingMethod::RMS:
220  default:
221  amplitude=RMSAmplitude(*dptr);
222  };
223  ++nlive;
224  amps.push_back(log(amplitude));
225  }
226  /*Silently return a 0 if there are no live data members*/
227  if(nlive==0) return 0.0;
228  mspass::utility::VectorStatistics<double> ampstats(amps);
229  if(use_mean)
230  {
231  avgamp=ampstats.mean();
232  }
233  else
234  {
235  avgamp=ampstats.median();
236  }
237 
238  /* restore to a value instead of natural log*/
239  avgamp=exp(avgamp);
240  double dscale=level/avgamp;
241  /* Now scale the data and apply calib */
242  for(dptr=d.member.begin();dptr!=d.member.end();++dptr)
243  {
244  if(dptr->live())
245  {
246  double calib;
247  (*dptr) *= dscale;
248  if(dptr->is_defined(scale_factor_key))
249  {
250  calib=dptr->get_double(scale_factor_key);
251  }
252  else
253  {
254  calib=1.0;
255  }
256  calib/=dscale;
257  dptr->put(scale_factor_key,calib);
258  }
259  }
260  return avgamp;
261  }catch(...){throw;};
262 }
